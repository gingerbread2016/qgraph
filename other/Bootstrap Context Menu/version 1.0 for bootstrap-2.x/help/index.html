3. Tutorial Top

To use Twitter bootstrap context menu, please add html code bellow into your website page. 

<link href="../bootstrap/css/bootstrap.min.css" rel="stylesheet">
<script src="../jquery/jquery-1.8.2.min.js"></script>
<script src="../js/bootstrap.contextmenu.js"></script>

Create a html tag which is registered the context menu (Twitter bootstrap context menu). 

<div id="id0">Right click here </div>

To create a Twitter bootstrap context menu, it is simple. Please see code bellow. 

var menu = {}; // define a x-context menu object.

menu['copy'] = {text:'Copy'}; // add Copy element to menu.
menu['paste'] = {text:'Paste'}; // add Paste element to menu.
menu['paste_shortcut'] = {text:'Paste shortcut'}; // add Paste shortcut element to menu.

$('#id0').contextMenu(menu); // register menu to a html tag

Add event to a Twitter bootstrap context menu element. each Twitter bootstrap context menu element have a callback function, when it is clicked, callback function is triggered. 

var menu = {}; // define a x-context menu object.

/**
	define a callback function
	target: is a html tag which is registered the Twitter bootstrap context menu.
	element: is a Twitter bootstrap context menu element is clicked.
*/ 
var callback = function(target,element){
			$(target).html('<span style="color:red">' +$(element).html() +'</span> menu is clicked,'+
			 'Target id: '+ $(target).attr('id'));
		};

menu['copy'] = {text:'Copy',click:callback}; // add Copy element to menu.
menu['paste'] = {text:'Paste',click:callback}; // add Paste element to menu.
menu['paste_shortcut'] = {text:'Paste shortcut',click:callback}; // add Paste shortcut element to menu.
$('#id0').data('mc',1);
$('#id0').contextMenu(menu); // register menu to a html tag

Enable/Disable Twitter bootstrap context menu element. Twitter bootstrap context menu element have a disabled property. If this property is set true, Twitter bootstrap context menu element is disable.
If this property is set false or not define, Twitter bootstrap context menu element is enable. 

var menu = {}; // define a x-context menu object.

/**
	define a callback function
	target: is a html tag which is registered the Twitter bootstrap context menu.
	element: is a Twitter bootstrap context menu element is clicked.
*/ 
var callback = function(target,element){
			$(target).html('<span style="color:red">' +$(element).html() +'</span> menu is clicked,'+
			 'Target id: '+ $(target).attr('id'));
		};

menu['copy'] = {text:'Copy',disabled:true,click:callback}; // this element is disabled
menu['paste'] = {text:'Paste',disabled:false,click:callback}; 
menu['paste_shortcut'] = {text:'Paste shortcut',click:callback}; 

$('#id0').contextMenu(menu);

other way enable/disable a Twitter bootstrap context menu element. 

/**
	first param is enable or disable
	second param is the key of Twitter bootstrap context menu element.
*/
$('#id0').contextMenu('enable','paste'); // enable paste element

/**
	first param is enable or disable
	second param is the key of Twitter bootstrap context menu element.
*/
$('#id0').contextMenu('disable','paste'); // disable paste element

Add submenu for the Twitter bootstrap context menu element. Twitter bootstrap context menu element have children property, this property is a Twitter bootstrap context menu. 

var menu = {}; // define a x-context menu object.

/**
	define a callback function
	target: is a html tag which is registered the Twitter bootstrap context menu.
	element: is a Twitter bootstrap context menu element is clicked.
*/ 
var callback = function(target,element){
			$(target).html('<span style="color:red">' +$(element).html() +'</span> menu is clicked,'+
			 'Target id: '+ $(target).attr('id'));
		};
// define submenu for sortby element.
menu['sortby']	= {text:'Sort by',click:callback,
			children:{
				name:{text:'Name',click:callback},
				size:{text:'Size',click:callback},
				itemtype:{text:'Item type',click:callback},
				datemodified:{text:'Date modified',click:callback},
				s1:'---',
				Ascending:{text:'Ascending',click:callback},
				Descending:{text:'Descending',click:callback},
				s2:'---',
				More:{text:'More',click:callback}
			}
		};
menu['copy'] = {text:'Copy',click:callback}; 
menu['paste'] = {text:'Paste',click:callback}; 
menu['s2'] = '---'; 
menu['paste_shortcut'] = {text:'Paste shortcut',click:callback}; 

$('#id0').contextMenu(menu);
